Q1.Find the Smaller Angle

        
      var Minimal_Angle = (h, m) => 
   {
  
   h=h%12;
   hh=(h*360)/12 +(m*360)/(12*60);
   mm=(m*360)/(60);
   var angle =Math.abs(hh-mm);
   if(angle>180){
     angle=360-angle;
   }
   return angle;
       
};

Q3.Perfect Number Check.



 
    var Perfect_Check = (n) => 
    {
       
    let ori=n;
    let sum=0;
    
    for(let i=0;i < n;i++){
      if(n%i==0){
        
      sum=sum+i;
       
       
      }
    }
    
    if(sum==ori){
      return "YES";
    }
    else{
      return "NO";
    }   
            
    };
    



Q4.Reverse a Number


     
var Reverse_Number = (n) => 
{
  let a=[];
  let r;
  
  while(n!=0){
    
    r=n%10
    a.push(r);
    n=parseInt(n/10);
  }
  
  
for(let i=0;i < a.length;i++){
  if(a[i]==0)
  {
    a.splice(i,1);
    i=i-1;

  }
  else{
    break;
  }

}

let result=a.join('')

return result;
};
